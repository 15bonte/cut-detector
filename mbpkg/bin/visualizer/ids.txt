### Globals (defined in app) ###
sig_open_pannel: dmc.Store <- ["imp", "det", "trk", "viz"]
    The pannel to open

sig_cur_frame: dmc.Store <- int
    The current frame number

sig_imp_file: dmc.Store <- {path: str, fmt: str}
    The file to import globally, written to once the file has been imported

sig_det_param: dmc.Store <- dict of parameters
    The parameters set for the detector
