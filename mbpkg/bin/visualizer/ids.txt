### Globals (defined in app) ###
sig_open_pannel: dcc.Store <- ["imp", "det", "trk", "viz"]
    The pannel to open

sig_cur_frame: dcc.Store <- int
    The current frame number

sig_imp_file: dcc.Store <- {path: str, fmt: str}
    The file to import globally, written to once the file has been imported

sig_det_param: dcc.Store <- dict of parameters
    The parameters set for the detector

sig_cur_det: dcc.Store <- str
    The str of the Detector. either a kwstr or a fmtstr.

sig_viz_param: dcc.Store <- {layer: str, args...}
    The visualization settings for the debugging (!= rendering parameters)


### SideBar ###
sb_imp_btn: dmc.ActionIcon
    Importation pannel

sb_det_btn: dmc.ActionIcon
    Detection pannel

sb_trk_btn: dmc.ActionIcon
    Tracking pannel

sb_viz_btn: dmc.ActionIcon
    Vizualization pannel

sb_all_btn: dmc.ActionIcon
    Merge all pannels into 1 big scollable pannel


### Main pannel ###
pannel_area: html.Div
    Where the pannel will be generated

@@ Imp pannel @@
imp_dir_sel: dmc.Select
    Select dirpath

imp_file_sel: dmc.Select
    Select filename

imp_fmt_type: dmc.Select
    Select format

imp_btn: dmc.Button
    Import button

@@ Det pannel @@
det_sel: dmc.Select [Detector.try_to_kwstr() or Detector.to_str()]
    The base detector selector

det_param_area: html.Div
    Area where the parameters appear

det_param: dmc.NumberInput (maybe use dash matching feature...)
    The detector parameters


@@ Viz pannel @@
viz_layer: dmc.Select
    Select the layer

viz_parameters: html.Div
    Where the paramters are shown

viz_param: dmc.NumberInput
    The visualization parameters that are generated at runtime.


### Viewer ###
graph: dcc.Graph
    The display for the image

vctrl_pbtn: dmc.Button
    Previous frame button

vctrl_nbtn: dmc.Button
    Next frame btn

vctrl_frame_input: dmc.NumberInput
    The current frame number input

vctr_frame_max_text: dmc.Text
    Max index for frames
